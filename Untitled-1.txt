name: Build & Deploy Angular to GitHub Pages

on:
  push:
    branches: [ main ]
  workflow_dispatch:

permissions:
  contents: read
  pages: write
  id-token: write

concurrency:
  group: pages-deploy
  cancel-in-progress: true

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm

      - name: Install dependencies
        run: npm ci

      - name: Detect Angular project name
        id: project
        run: |
          PROJECT_NAME=$(node -p "require('./angular.json').defaultProject || Object.keys(require('./angular.json').projects)[0]")
          echo "project=$PROJECT_NAME" >> "$GITHUB_OUTPUT"

      - name: Build (project page)
        if: ${{ github.event.repository.name != format('{0}.github.io', github.repository_owner) }}
        run: |
          npx ng build --configuration production --base-href "/${{ github.event.repository.name }}/"
        env:
          NODE_OPTIONS: --max_old_space_size=4096

      - name: Build (user/organization root page)
        if: ${{ github.event.repository.name == format('{0}.github.io', github.repository_owner) }}
        run: |
          npx ng build --configuration production --base-href "/"
        env:
            NODE_OPTIONS: --max_old_space_size=4096

      - name: Add 404.html for SPA refresh support
        run: |
          DIST_DIR="dist/${{ steps.project.outputs.project }}"
          cp "$DIST_DIR/index.html" "$DIST_DIR/404.html"

      # (Optional) Uncomment to run tests; ensure a headless browser setup if you enable.
      # - name: Test
      #   run: npm test -- --watch=false --browsers=ChromeHeadless

      - name: Upload artifact
        uses: actions/upload-pages-artifact@v3
        with:
          path: dist/${{ steps.project.outputs.project }}

  deploy:
    needs: build
    runs-on: ubuntu-latest
    environment:
      name: github-pages
      url: ${{ steps.deploy.outputs.page_url }}
    steps:
      - name: Deploy to GitHub Pages
        id: deploy
        uses: actions/deploy-pages@v4
